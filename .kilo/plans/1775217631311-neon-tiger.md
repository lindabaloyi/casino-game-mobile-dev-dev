```javascript
_isSocketValid(socketEntry) {
  if (!socketEntry || !socketEntry.socket) {
    this._log('WARN', 'Invalid socket entry structure', {
      socketEntry: socketEntry,
      hasSocket: !!socketEntry?.socket
    });
    return false;
  }
  
  const isConnected = socketEntry.socket.connected === true;
  const isDisconnected = socketEntry.socket.disconnected === false;
  const isValid = isConnected && isDisconnected;
  
  this._log('DEBUG', 'Socket validation result', {
    socketId: socketEntry.id,
    userId: socketEntry.userId,
    connected: socketEntry.socket.connected,
    disconnected: socketEntry.socket.disconnected,
    valid: isValid
  });
  
  return isValid;
}
```

### 3. Cleanup Queue Logging
**Location:** `_cleanupQueue()` method

Add before/after state logging:

```javascript
_cleanupQueue(gameType) {
  const queue = this.waitingQueues[gameType];
  if (!queue) {
    this._log('WARN', 'Attempted cleanup on non-existent queue', { gameType });
    return;
  }
  
  const originalLength = queue.length;
  this._log('DEBUG', 'Starting queue cleanup', {
    gameType,
    originalCount: originalLength,
    socketIds: queue.map(e => e.id),
    userIds: queue.map(e => e.userId)
  });
  
  const removedEntries = [];
  this.waitingQueues[gameType] = queue.filter(entry => {
    if (this._isSocketValid(entry)) return true;
    
    removedEntries.push({
      socketId: entry.id,
      userId: entry.userId,
      wasInGameMap: this.socketGameMap.has(entry.id)
    });
    
    if (this.socketGameMap.has(entry.id)) {
      this.socketGameMap.delete(entry.id);
    }
    return false;
  });
  
  const newLength = this.waitingQueues[gameType].length;
  const removedCount = originalLength - newLength;
  
  if (removedCount > 0) {
    this._log('INFO', 'Cleaned stale sockets from queue', {
      gameType,
      removedCount,
      originalCount: originalLength,
      newCount: newLength,
      removedEntries
    });
  } else {
    this._log('DEBUG', 'No stale sockets found in queue', {
      gameType,
      queueSize: newLength
    });
  }
}
```

```javascript
    afterCounts,
    totalCleaned: Object.keys(beforeCounts).reduce((sum, type) =>

  });
```

### 5. Disconnection Handling Logging
**Location:** `handleDisconnection()` method

Add comprehensive disconnection logs:

```javascript
handleDisconnection(socket) {
  this._log('INFO', 'Handling socket disconnection', {
    socketId: socket.id,
    userId: socket.userId,
    socketConnected: socket.connected,
    socketDisconnected: socket.disconnected
  });
  
  const socketInfo = this.socketGameMap.get(socket.id);
  if (!socketInfo) {
    this._log('WARN', 'Socket not found in game map on disconnect', {
      socketId: socket.id,
      userId: socket.userId
    });
    return null;
  }

  const { gameId, gameType, userId } = socketInfo;
  this._log('INFO', 'Socket found in game map', {
    socketId: socket.id,
    gameId,
    gameType,
    userId,
    wasInQueue: !gameId
  });

  if (!gameType || !this.waitingQueues[gameType]) {
    this._log('ERROR', 'Invalid game type on disconnect', {
      socketId: socket.id,
      gameType,
      availableTypes: Object.keys(this.waitingQueues)
    });
    this.socketGameMap.delete(socket.id);
    return null;
  }

  if (!gameId) {
    // Remove from waiting queue
    const queueBefore = this.waitingQueues[gameType].length;
    this.waitingQueues[gameType] = this.waitingQueues[gameType].filter(s => s.id !== socket.id);
    const queueAfter = this.waitingQueues[gameType].length;
    
    this._log('INFO', 'Removed disconnected socket from waiting queue', {
      socketId: socket.id,
      userId,
      gameType,
      queueBefore,
      queueAfter,
      removed: queueBefore - queueAfter
    });
    
    this.socketGameMap.delete(socket.id);
    return null;
  }

  // Handle in-game disconnection
  this.socketGameMap.delete(socket.id);
  const sockets = (this.gameSocketsMap.get(gameId) || []).filter(id => id !== socket.id);

  this._log('INFO', 'Player disconnected from active game', {
    socketId: socket.id,
    userId,
    gameId,
    gameType,
    remainingSockets: sockets.length,
    gameEnding: sockets.length === 0
  });

  if (sockets.length === 0) {
    this.gameSocketsMap.delete(gameId);
    this.gameManager.endGame(gameId);
    this._log('INFO', 'Game ended due to all players disconnecting', { gameId });
  } else {
    this.gameSocketsMap.set(gameId, sockets);
  }

  return { gameId, remainingSockets: sockets };
}
```

### 6. Queue Management Logging
**Location:** `addToQueue()` and `leaveQueue()` methods

Enhance existing logs with timestamps and more details:

```javascript
addToQueue(socket, gameType, userId = null) {
  this._log('INFO', 'Adding player to queue', {
    socketId: socket.id,
    userId,
    gameType,
    socketConnected: socket.connected,
    socketDisconnected: socket.disconnected
  });
  
  if (this.socketGameMap.has(socket.id)) {
    this._log('WARN', 'Socket already in queue/game, rejecting', {
      socketId: socket.id,
      userId,
      existingInfo: this.socketGameMap.get(socket.id)
    });
    return null;
  }

  const socketEntry = { id: socket.id, socket, userId };
  this.waitingQueues[gameType].push(socketEntry);
  this.socketGameMap.set(socket.id, { gameId: null, gameType, userId });

  this._log('INFO', 'Successfully added to queue', {
    socketId: socket.id,
    userId,
    gameType,
    newQueueSize: this.waitingQueues[gameType].length
  });
  
  return this._tryCreateGame(gameType);
}

leaveQueue(socket, gameType) {
  this._log('INFO', 'Player explicitly leaving queue', {
    socketId: socket.id,
    userId: socket.userId,
    gameType
  });
  
  if (!gameType || !this.waitingQueues[gameType]) {
    this._log('ERROR', 'Invalid game type for leave queue', {
      socketId: socket.id,
      userId: socket.userId,
      requestedGameType: gameType,
      availableTypes: Object.keys(this.waitingQueues)
    });
    return;
  }
  
  const queueBefore = this.waitingQueues[gameType].length;
  this.waitingQueues[gameType] = this.waitingQueues[gameType].filter(entry => entry.id !== socket.id);
  const queueAfter = this.waitingQueues[gameType].length;
  
  this.socketGameMap.delete(socket.id);
  
  this._log('INFO', 'Player left queue', {
    socketId: socket.id,
    userId: socket.userId,
    gameType,
    queueBefore,
    queueAfter,
    removed: queueBefore - queueAfter
  });
}
```
